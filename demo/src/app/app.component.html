@Component({
  selector: 'app-root',

  template: `
    <ngx-dual-datepicker

    </ngx-dual-datepicker>
  `
})
export class AppComponent {
  onDateRangeChange(range: DateRange) {
    console.log('Selected range:', range);
  }
}

  // ‚ö° Option 1: Get all common presets (15+ presets)

  // üéØ Option 2: Get specific preset by ID
  lastWeekRange = this.engine.getPreset('lastWeek');
  // Returns: { startDate: '2026-02-14', endDate: '2026-02-20' }
  
  // üé® Option 3: Build custom preset list
  customPresets: PresetConfig[] = [
    { label: 'Last Week', getValue: () => this.engine.getPreset('lastWeek') },
    { label: 'Last Month', getValue: () => this.engine.getPreset('lastMonth') },
    { label: 'Last Quarter', getValue: () => this.engine.getPreset('lastQuarter') },

  ];
}

üèóÔ∏è Headless Architecture v3.5.0

Version 3.5.0 introduces a headless architecture that separates date range logic from UI. Build custom date pickers while leveraging battle-tested date logic, or use the core primitives in services, utilities, and business logic.

Core Components

üì¶ DualDateRangeStore

Injectable service with Angular Signals for reactive state management.

import { DualDateRangeStore } from '@oneluiz/dual-datepicker';

@Component({
  standalone: true,
  providers: [DualDateRangeStore]  // Provide at component level
})
export class CustomDatePicker {
  store = inject(DualDateRangeStore);
  
  // Read reactive state
  currentRange = this.store.range;        // Signal<DateRange>
  isValid = this.store.isValidRange;      // Signal<boolean>
  hasChanges = this.store.hasChanges;     // Signal<boolean>
  
  // Modify state
  selectDate(date: string) {
    this.store.setStart(date);
    this.store.setEnd(date);
    this.store.applyPending();
  }
}

Key Methods:

  • setStart(date: string) - Set start date
  • setEnd(date: string) - Set end date
  • setRange(start: string, end: string) - Set both dates
  • applyPending() - Confirm pending changes
  • cancelPending() - Revert to last applied range
  • clear() - Clear all dates
  • applyPreset(presetId: string) - Apply a preset
  • configure(options) - Update configuration
